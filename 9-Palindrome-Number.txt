impl Solution {
    pub fn is_palindrome(x: i32) -> bool {
        if x < 0
        {
            return false
        }
        else
        {
            let xStr = x.to_string();
            let xStrRev = x.to_string().chars().rev().collect::<String>();
            xStr.eq(&xStrRev)
        }
    }
}
